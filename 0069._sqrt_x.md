class Solution(object):
    def mySqrt(self, x):
        """
        :type x: int
        :rtype: int
        """
        import math
        result = int(math.sqrt(x))
        return result

# Solution 2 二分查找
求n的平方根的整数部分，所以平方根一定是 1，2，3 ... n 中的一个数。从一个有序序列中找一个数，就可以用二分查找。
```python
class Solution:
    def mySqrt(self, x):
        low = 0
        high = x

        while low < high:
            mid = (low + high) // 2

            if mid ** 2 <= x < (mid+1) ** 2:
                return mid
            elif mid ** 2 < x:
                low = mid + 1
            else:
                high = mid - 1

        return low
```

```python
class Solution:
    def mySqrt(self, x):
        low = 0
        high = x

        while low <= high:
            mid = (low + high) // 2

            if mid * mid > x:
                high = mid - 1
            elif mid * mid < x:
                low = mid + 1
            else:
                return mid
        # 如果没有完美的平方根，那么high就在完美平方根的左边，left就在完美平方根的右边
        # 例如x = 2 -> 0, 1, 2中找平方根。
        return high
```
