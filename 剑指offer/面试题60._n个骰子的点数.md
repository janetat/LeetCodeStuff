# Solution 1 卷积
参考别人的答案
```python
import numpy as np


class Solution:
    def twoSum(self, n: int) -> List[float]:
        one_dice = np.array([1/6 for _ in range(6)])
        res = np.array([1/6 for _ in range(6)])
        
        for i in range(1, n):  # repeat n - 1 time
            res = np.convolve(res, one_dice)

        return res
```
